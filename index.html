} function render(id, points, color, label, unit, progress) { const c = document.getElementById(id); const ctx = c.getContext('2d'); const dpr = window.devicePixelRatio || 2; c.width = c.clientWidth * dpr; c.height = c.clientHeight * dpr; const w = c.width, h = c.height; const pL = w * 0.1, pR = w * 0.1, pT = h * 0.1, pB = h * 0.1; const gW = w - pL - pR, gH = h - pT - pB; const center = h / 2; ctx.clearRect(0, 0, w, h); let maxAbs = Math.max(...points.map(Math.abs)); if (maxAbs < 0.1) maxAbs = 1; const scaleY = (gH / 2) / maxAbs; // Raster & Assen ctx.strokeStyle = "#e0e0e0"; ctx.lineWidth = 1 * dpr; for (let i = -4; i <= 4; i++) { const y = center - ((maxAbs / 4) * i * scaleY); ctx.beginPath(); ctx.moveTo(pL, y); ctx.lineTo(w - pR, y); ctx.stroke(); } ctx.strokeStyle = "#bbb"; ctx.lineWidth = 2 * dpr; ctx.beginPath(); ctx.moveTo(pL, center); ctx.lineTo(w - pR, center); ctx.stroke(); // Data lijn ctx.beginPath(); ctx.strokeStyle = color; ctx.lineWidth = 4 * dpr; ctx.lineJoin = "round"; points.forEach((p, i) => { const x = pL + (i / (points.length - 1)) * gW; const y = center - (p * scaleY); i === 0 ? ctx.moveTo(x, y) : ctx.lineTo(x, y); }); ctx.stroke(); // ORANJE LOPER (Verticale lijn) if (progress > 0) { const cursorX = pL + (progress * gW); const dataIdx = Math.min(Math.floor(progress * points.length), points.length - 1); const cursorY = center - (points[dataIdx] * scaleY); ctx.strokeStyle = "#ff9500"; ctx.lineWidth = 3 * dpr; ctx.beginPath(); ctx.moveTo(cursorX, pT); ctx.lineTo(cursorX, h - pB); ctx.stroke(); // Bolletje op de lijn ctx.fillStyle = "#ff9500"; ctx.beginPath(); ctx.arc(cursorX, cursorY, 8 * dpr, 0, Math.PI * 2); ctx.fill(); ctx.strokeStyle = "#fff"; ctx.lineWidth = 2 * dpr; ctx.stroke(); } // Label ctx.fillStyle = "#333"; ctx.font = "bold " + (18 * dpr) + "px Arial"; ctx.textAlign = "left"; ctx.fillText(label + " (" + unit + ")", 20, 35 * dpr); }
