#include #include // ===================== // BLE UUIDs // ===================== BLEService castService("19b10000-e8f2-537e-4f6c-d104768a1214"); BLECharacteristic dataChar( "19b10001-e8f2-537e-4f6c-d104768a1214", BLERead | BLENotify, 12 // 6 x int16_t ); // ===================== // SETUP // ===================== void setup() { Serial.begin(115200); if (!IMU.begin()) { Serial.println("IMU FAIL"); while (1); } if (!BLE.begin()) { Serial.println("BLE FAIL"); while (1); } BLE.setLocalName("Cast"); BLE.setDeviceName("Cast"); BLE.setAdvertisedService(castService); castService.addCharacteristic(dataChar); BLE.addService(castService); BLE.advertise(); Serial.println("READY"); } // ===================== // LOOP // ===================== void loop() { BLEDevice central = BLE.central(); if (!central) return; Serial.println("CONNECTED"); delay(300); unsigned long lastMillis = 0; while (central.connected()) { if (millis() - lastMillis >= 20) { // 50 Hz lastMillis = millis(); float ax, ay, az, gx, gy, gz; if (IMU.accelerationAvailable() && IMU.gyroscopeAvailable()) { IMU.readAcceleration(ax, ay, az); IMU.readGyroscope(gx, gy, gz); int16_t packet[6]; packet[0] = (int16_t)(ax * 100); // m/s² * 100 packet[1] = (int16_t)(ay * 100); packet[2] = (int16_t)(az * 100); packet[3] = (int16_t)(gx); // dps packet[4] = (int16_t)(gy); packet[5] = (int16_t)(gz); dataChar.writeValue((byte*)packet, 12); } } delay(1); } Serial.println("DISCONNECTED"); }
